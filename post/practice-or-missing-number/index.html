Practice | Missing number

2021-04-26 Algorithm

Given a positive integer n(n≤40), pick n-1 numbers randomly from 1 to n and concatenate them in random order as a string s, which means there is a missing number between 1 and n. Can you find the missing number?(Notice that in some cases the answer will not be unique, and in these cases you only need to find one valid answer.）

Sample Input 1:

20
81971112205101569183132414117

Sample Output 1:

16

Solution:

#include <bits/stdc++.h>

using namespace std;

int len,n;
bool f[100];
string s;
int num[100];

void dfs(int k)
{
if(k>=len)
{
for(int i=1;i<=n;i++)
{
if(f[i]==false)
{
printf("%d\n",i);
}
}
return;
}
if(f[num[k]]==false)
{
f[num[k]]=true;
dfs(k+1);
f[num[k]]=false;
}
if(f[num[k]*10+num[k+1]]==false&&num[k]*10+num[k+1]<=n)
{
f[num[k]*10+num[k+1]]=true;
dfs(k+2);
f[num[k]*10+num[k+1]]=false;
}
}

int main()
{
scanf("%d",&n);
cin>>s;
for(int i=0;i<s.size();i++)
{
num[i]=s[i]-'0';
}
len=s.size();
dfs(0);
return 0;
}
