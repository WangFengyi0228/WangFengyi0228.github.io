Practice | Count number of binary strings without consecutive 1’s

2021-03-16 Algorithm

Given a positive integer n(3≤n≤90), count all possible distinct binary strings of length n such that there are no consecutive 1's .

Sample Input 1:

2

Sample Output 1:

3 // The 3 strings are 00, 01, 10

Sample Input 2:

3

Sample Output 2:

5 // The 5 strings are 000, 001, 010, 100, 101

Solution:

#include <bits/stdc++.h>

using namespace std;

int pao(int n)
{
    if(n==1)
    {
        return 2;
    }
    else if(n==2)
    {
        return 3;
    }
    else
    {
        return pao(n-1)+pao(n-2);
    }
}

int main()
{
    int n;
    scanf("%d",&n);
    int dp[100][2];
    dp[1][0]=1;
    dp[1][1]=1;
    for(int i=2;i<=n;i++)
    {
        dp[i][0]=dp[i-1][0]+dp[i-1][1];
        dp[i][1]=dp[i-1][0];
    }
    printf("%d\n",dp[n][0]+dp[n][1]);
    printf("%d\n",pao(n));
    return 0;
}
